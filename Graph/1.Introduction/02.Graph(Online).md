# 3. BFS

```java
public static void bfs(int start,
                       List<List<Integer>> graph){

    boolean[] visited =
            new boolean[graph.size()];

    Queue<Integer> q =
            new LinkedList<>();

    q.offer(start);
    visited[start] = true;

    while(!q.isEmpty()){

        int node = q.poll();

        System.out.print(node + " ");

        for(int nbr : graph.get(node)){

            if(!visited[nbr]){

                visited[nbr] = true;
                q.offer(nbr);
            }
        }
    }
}
```

Complexity:

```text
O(V + E)
```

# 5. Cycle Detection (Undirected)

```java
public static boolean dfs(int node,
                          int parent,
                          boolean[] visited,
                          List<List<Integer>> graph){

    visited[node] = true;

    for(int nbr : graph.get(node)){

        if(!visited[nbr]){

            if(dfs(nbr,node,visited,graph))
                return true;

        }else if(nbr != parent){

            return true;
        }
    }

    return false;
}
```

# 6. Cycle Detection (Directed)

```java
boolean[] visited;
boolean[] path;

public boolean dfs(int node,
                   List<List<Integer>> graph){

    visited[node] = true;
    path[node] = true;

    for(int nbr : graph.get(node)){

        if(!visited[nbr]){

            if(dfs(nbr,graph))
                return true;
        }
        else if(path[nbr]){
            return true;
        }
    }

    path[node] = false;

    return false;
}
```

# 7. Topological Sort (Kahn Algorithm)

```java
public List<Integer> topoSort(int n,
                              List<List<Integer>> graph){

    int[] indegree = new int[n];

    for(int i=0;i<n;i++){

        for(int nbr : graph.get(i)){
            indegree[nbr]++;
        }
    }

    Queue<Integer> q =
            new LinkedList<>();

    for(int i=0;i<n;i++){

        if(indegree[i]==0)
            q.offer(i);
    }

    List<Integer> ans =
            new ArrayList<>();

    while(!q.isEmpty()){

        int node = q.poll();

        ans.add(node);

        for(int nbr : graph.get(node)){

            indegree[nbr]--;

            if(indegree[nbr]==0)
                q.offer(nbr);
        }
    }

    return ans;
}
```

# 8. Shortest Path (Dijkstra)

```java
class Pair{
    int node;
    int dist;

    Pair(int node,int dist){
        this.node=node;
        this.dist=dist;
    }
}

PriorityQueue<Pair> pq =
    new PriorityQueue<>(
        (a,b)->a.dist-b.dist);
```

```java
int[] dist = new int[n];

Arrays.fill(dist,Integer.MAX_VALUE);

dist[src] = 0;

pq.offer(new Pair(src,0));

while(!pq.isEmpty()){

    Pair curr = pq.poll();

    int node = curr.node;
    int d = curr.dist;

    for(Pair nbr : graph.get(node)){

        if(d + nbr.dist <
           dist[nbr.node]){

            dist[nbr.node]
              = d + nbr.dist;

            pq.offer(
              new Pair(
                 nbr.node,
                 dist[nbr.node]
              )
            );
        }
    }
}
```

# 9. MST (Kruskal)

### DSU

```java
int[] parent;
int[] rank;

public int find(int x){

    if(parent[x]==x)
        return x;

    return parent[x]
           = find(parent[x]);
}
```

```java
public void union(int x,int y){

    int px = find(x);
    int py = find(y);

    if(px==py) return;

    if(rank[px]<rank[py]){
        parent[px]=py;
    }
    else if(rank[px]>rank[py]){
        parent[py]=px;
    }
    else{
        parent[py]=px;
        rank[px]++;
    }
}
```

### Kruskal

```java
Arrays.sort(edges,
   (a,b)->a[2]-b[2]);

int mstCost = 0;

for(int[] edge : edges){

    int u = edge[0];
    int v = edge[1];
    int w = edge[2];

    if(find(u)!=find(v)){

        union(u,v);

        mstCost += w;
    }
}
```
